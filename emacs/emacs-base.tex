\subsection{常见命令}
\label{sec:org3492d86}
\begin{itemize}
\item M-x 进入命令行
\begin{itemize}
\item C-g 取消命令
\item C-h 帮助文档
\item C-x-q 设置文件只读，重复可取消
\end{itemize}

\item M-x 可以输入函数命令
\begin{itemize}
\item C-x-s 保存文件
\item C-x-c 退出emacs
\item C-x b 切换buffer,输入名称打开
\item C-x-b 打开buffer列表
\item C-x d 打开目录，可以跳转打开目录
\end{itemize}
\end{itemize}


\begin{itemize}
\item buffer list标记操作
\begin{itemize}
\item s 添加save标记
\item d或者C-k 标记关闭标记
\item o，C-o 打开buffer
\item 1 当前widow打开buffer，evil下无效；
\item x 执行标记操作
\end{itemize}
\end{itemize}


\begin{itemize}
\item 窗口操作
\begin{itemize}
\item C-x 0 关闭当前窗口
\item C-x 1 只保留当前窗口
\item 窗口大小调节 \textbf{\textbf{C-x \^{}}} 加高当前窗口
\item \textbf{\textbf{C-x \{**变窄，**C-x \}}} 变宽
\end{itemize}
\end{itemize}
